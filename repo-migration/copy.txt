param(
    [Parameter(Mandatory=$true)] [string]$TfsUrl,
    [Parameter(Mandatory=$true)] [string]$GitHubOrg,
    [Parameter(Mandatory=$true)] [string]$JsonFilePath
)

# Read tokens from environment variables
$TfsToken = $env:TFS_TOKEN
$GitHubToken = $env:GITHUB_TOKEN

if (-not $TfsToken) { throw "TFS_TOKEN is not set in environment variables." }
if (-not $GitHubToken) { throw "GITHUB_TOKEN is not set in environment variables." }

# Define working directory
$workingDir = Join-Path -Path $PWD -ChildPath "repo"

Write-Host "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - üìÅ Starting migration as per $JsonFilePath"

# Read and parse JSON file
if (-not (Test-Path $JsonFilePath)) {
    throw "JSON file $JsonFilePath not found."
}
$jsonContent = Get-Content -Raw -Path $JsonFilePath | ConvertFrom-Json

foreach ($project in $jsonContent.projects) {
    $projectName = $project.projectName
    foreach ($repo in $project.repositories) {
        $repoName = $repo.name
        $repoType = $repo.type.ToUpper()

        $tfvcPath = "$/$projectName/$repoName"
        Write-Host "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - üìÇ Project: $projectName, Repo: $repoName, Type: $repoType"
        Write-Host "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - üìå Current Working Directory: $workingDir"

        # Create local repo folder path
        $localRepoPath = Join-Path -Path $workingDir -ChildPath $repoName

        # Step 1: Create GitHub repo via API
        Write-Host "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - üîß Creating GitHub repo: $repoName"

        $createRepoUri = "https://api.github.com/orgs/$GitHubOrg/repos"
        $repoBody = @{
            name = $repoName
            private = $true
            auto_init = $false
        } | ConvertTo-Json

        $headers = @{
            Authorization = "token $GitHubToken"
            Accept = "application/vnd.github.v3+json"
            "User-Agent" = "PowerShell-Script"
        }

        try {
            $response = Invoke-RestMethod -Uri $createRepoUri -Method Post -Headers $headers -Body $repoBody
            Write-Host "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - ‚úÖ Created GitHub repo: $repoName"
        } catch {
            Write-Warning "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - ‚ö†Ô∏è Failed to create GitHub repo: $repoName. Error: $_"
            continue
        }

        if ($repoType -eq "TFVC") {
            # Clone TFVC repo using git-tfs
            Write-Host "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - üì• Cloning TFVC repo using git-tfs"

            if (Test-Path $localRepoPath) {
                Remove-Item -Recurse -Force -Path $localRepoPath
            }

            $cloneCmd = "git tfs clone --username user --password $TfsToken $TfsUrl $tfvcPath $localRepoPath"
            Write-Host "Running: $cloneCmd"
            $cloneResult = & git tfs clone --username user --password $TfsToken $TfsUrl $tfvcPath $localRepoPath
            if ($LASTEXITCODE -ne 0) {
                Write-Warning "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - ‚ùå git-tfs clone failed for repo $repoName"
                continue
            }

            Set-Location -Path $localRepoPath
            git remote add origin "https://$GitHubToken@github.com/$GitHubOrg/$repoName.git"
            git push origin --all
            git push origin --tags
            Set-Location -Path $workingDir

            Remove-Item -Recurse -Force -Path $localRepoPath
            Write-Host "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - üóëÔ∏è Deleted local repo directory: $localRepoPath"
        }
        elseif ($repoType -eq "GIT") {
            # For Git repos, you might want to clone from TFS Git repo and push to GitHub
            # Add your Git repo migration logic here if needed
            Write-Host "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - ‚ÑπÔ∏è Git repo migration logic not implemented for repo $repoName"
        }
        else {
            Write-Warning "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - ‚ö†Ô∏è Unknown repo type '$repoType' for repo $repoName"
        }
    }
}

Write-Host "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') - üöÄ Migration completed."
